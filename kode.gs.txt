function doGet(e) {
  // Health check endpoint
  return jsonResponse({ status: "success", message: "FinPaws API is Running!" });
}

function doPost(e) {
  try {
    const action = (e && e.parameter && e.parameter.action) ? e.parameter.action : "";

    if (action === "doLogin") {
      const user = (e.parameter.user || "").trim();
      const pass = (e.parameter.pass || "").trim();
      return jsonResponse(doLogin(user, pass));
    }

    if (action === "signUp") {
      const user = (e.parameter.user || "").trim();
      const pass = (e.parameter.pass || "").trim();
      return jsonResponse(signUp(user, pass));
    }

    if (action === "saveUserSetup") {
      const username = (e.parameter.username || "").trim();
      const mascotRaw = e.parameter.mascot || "";
      let mascotObj = null;

      try {
        mascotObj = JSON.parse(mascotRaw);
      } catch (err) {
        return jsonResponse({ status: "error", message: "Format mascot tidak valid." });
      }

      return jsonResponse(saveUserSetup(username, mascotObj));
    }

    if (action === "saveTransaction") {
      const username = (e.parameter.username || "").trim();
      const txRaw = e.parameter.txData || "";
      let txObj = null;

      try {
        txObj = JSON.parse(txRaw);
      } catch (err) {
        return jsonResponse({ status: "error", message: "Format transaksi tidak valid." });
      }

      return jsonResponse(saveTransaction(username, txObj));
    }

    if (action === "saveReminder") {
      const username = (e.parameter.username || "").trim();
      const reminderRaw = e.parameter.reminderData || "";
      let reminderObj = null;

      try {
        reminderObj = JSON.parse(reminderRaw);
      } catch (err) {
        return jsonResponse({ status: "error", message: "Format reminder tidak valid." });
      }

      return jsonResponse(saveReminder(username, reminderObj));
    }

    if (action === "completeReminder") {
      const username = (e.parameter.username || "").trim();
      const reminderRaw = e.parameter.reminderData || "";
      let reminderObj = null;

      try {
        reminderObj = JSON.parse(reminderRaw);
      } catch (err) {
        return jsonResponse({ status: "error", message: "Format reminder selesai tidak valid." });
      }

      return jsonResponse(completeReminder(username, reminderObj));
    }

    return jsonResponse({ status: "error", message: "Action tidak dikenali." });
  } catch (err) {
    return jsonResponse({ status: "error", message: "Server error: " + err.message });
  }
}

function jsonResponse(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function setupDatabase() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  
  // 1. Sheet Users
  let userSheet = ss.getSheetByName("Users");
  if (!userSheet) {
    userSheet = ss.insertSheet("Users");
    userSheet.appendRow(["user_id", "username", "password", "mascot_json"]);
    userSheet.setFrozenRows(1);
    userSheet.getRange("A1:D1").setFontWeight("bold").setBackground("#6c5ce7").setFontColor("white");
    
    // Insert dummy user (Admin & Peserta Baru)
    userSheet.appendRow(["U-001", "admin", "edudigital", JSON.stringify({hewan:"anjing", ikon:"🐶", tema:"theme-anjing", nama:"Doggo"})]);
    userSheet.appendRow(["U-002", "peserta", "edudigital", ""]); // Mascot kosong = butuh setup
  }

  // 2. Sheet Transactions
  let txSheet = ss.getSheetByName("Transactions");
  if (!txSheet) {
    txSheet = ss.insertSheet("Transactions");
    txSheet.appendRow(["trans_id", "username", "tanggal", "tipe", "kategori", "nominal", "keterangan", "foto_base64"]);
    txSheet.setFrozenRows(1);
    txSheet.getRange("A1:H1").setFontWeight("bold").setBackground("#6c5ce7").setFontColor("white");
  }

  // 3. Sheet Wishlists & Reminders (Opsional, untuk pengembangan fitur simpan cloud)
  let wishSheet = ss.getSheetByName("Wishlists");
  if (!wishSheet) {
    wishSheet = ss.insertSheet("Wishlists");
    wishSheet.appendRow(["wish_id", "username", "nama_target", "target_nominal", "terkumpul"]);
    wishSheet.setFrozenRows(1);
  }

  let remSheet = ss.getSheetByName("Reminders");
  if (!remSheet) {
    remSheet = ss.insertSheet("Reminders");
    remSheet.appendRow(["rem_id", "username", "judul", "tipe", "jatuh_tempo", "status", "recurring", "recurring_type", "recurring_days", "completed_count", "on_time_count", "last_completed_at"]);
    remSheet.setFrozenRows(1);
  } else {
    ensureReminderColumns(remSheet);
  }

  return "Setup Sukses! Sheet sudah dibuat.";
}

function doLogin(username, password) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const userSheet = ss.getSheetByName("Users");
  
  if (!userSheet) return { status: 'error', message: 'Database belum di-setup. Jalankan setupDatabase() di GAS.' };

  const data = userSheet.getDataRange().getValues();
  // Loop data dari baris ke-2 (skip header)
  for (let i = 1; i < data.length; i++) {
    if (data[i][1] === username && data[i][2] === password) {
      
      let mascotData = null;
      if (data[i][3]) {
        try { mascotData = JSON.parse(data[i][3]); } catch (e) { }
      }

      return {
        status: 'success',
        data: {
          username: data[i][1],
          isNew: mascotData === null,
          mascot: mascotData
        }
      };
    }
  }
  return { status: 'error', message: 'Username atau Password salah ngab!' };
}

function signUp(username, password) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const userSheet = ss.getSheetByName("Users");

  if (!userSheet) {
    return { status: 'error', message: 'Database belum di-setup. Jalankan setupDatabase() di GAS.' };
  }

  if (!username || username.length < 3) {
    return { status: 'error', message: 'Username minimal 3 karakter.' };
  }

  if (!password || password.length < 4) {
    return { status: 'error', message: 'Password minimal 4 karakter.' };
  }

  const data = userSheet.getDataRange().getValues();
  const usernameLower = username.toLowerCase();
  let maxNumericId = 0;

  for (let i = 1; i < data.length; i++) {
    const existingUsername = String(data[i][1] || "").toLowerCase();
    if (existingUsername === usernameLower) {
      return { status: 'error', message: 'Username sudah dipakai, pilih yang lain.' };
    }

    const userId = String(data[i][0] || "");
    const match = userId.match(/^U-(\d+)$/);
    if (match) {
      const numeric = Number(match[1]);
      if (numeric > maxNumericId) maxNumericId = numeric;
    }
  }

  const newId = "U-" + String(maxNumericId + 1).padStart(3, '0');
  userSheet.appendRow([newId, username, password, ""]);

  return {
    status: 'success',
    data: {
      username: username,
      isNew: true,
      mascot: null
    }
  };
}

function saveUserSetup(username, mascotObj) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const userSheet = ss.getSheetByName("Users");
  const data = userSheet.getDataRange().getValues();
  
  for (let i = 1; i < data.length; i++) {
    if (data[i][1] === username) {
      // Update cell mascot_json
      userSheet.getRange(i + 1, 4).setValue(JSON.stringify(mascotObj));
      return { status: 'success' };
    }
  }
  return { status: 'error', message: 'User tidak ditemukan' };
}

// Fungsi dummy untuk menyimpan transaksi dari frontend ke sheet
function saveTransaction(username, txData) {
  if (!username) return { status: 'error', message: 'Username wajib diisi' };
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const txSheet = ss.getSheetByName("Transactions");
  if (!txSheet) return { status: 'error', message: 'Sheet Transactions tidak ditemukan' };
  
  txSheet.appendRow([
    txData.id,
    username,
    txData.tgl,
    txData.tipe,
    txData.kategori,
    txData.nominal,
    txData.ket,
    txData.fotoUrl || ""
  ]);
  
  return { status: 'success' };
}

function ensureReminderColumns(remSheet) {
  const required = [
    "rem_id", "username", "judul", "tipe", "jatuh_tempo", "status", "recurring",
    "recurring_type", "recurring_days", "completed_count", "on_time_count", "last_completed_at"
  ];

  const headerRange = remSheet.getRange(1, 1, 1, Math.max(remSheet.getLastColumn(), 1));
  const existing = headerRange.getValues()[0].map(String);

  if (existing.length === 1 && existing[0] === "") {
    remSheet.getRange(1, 1, 1, required.length).setValues([required]);
    remSheet.setFrozenRows(1);
    return;
  }

  let appendFrom = existing.length + 1;
  for (var i = existing.length; i < required.length; i++) {
    remSheet.getRange(1, appendFrom).setValue(required[i]);
    appendFrom++;
  }
}

function saveReminder(username, reminderData) {
  if (!username) return { status: 'error', message: 'Username wajib diisi' };

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const remSheet = ss.getSheetByName("Reminders");
  if (!remSheet) return { status: 'error', message: 'Sheet Reminders tidak ditemukan' };

  ensureReminderColumns(remSheet);

  const data = remSheet.getDataRange().getValues();
  const remId = String(reminderData.id || "");
  if (!remId) return { status: 'error', message: 'Reminder ID wajib ada' };

  const rowValues = [
    remId,
    username,
    reminderData.judul || "",
    reminderData.tipe || "Tagihan",
    reminderData.tgl || "",
    reminderData.status || "open",
    reminderData.recurring ? "TRUE" : "FALSE",
    reminderData.recurringType || "monthly",
    reminderData.recurringDays || "",
    reminderData.completedCount || 0,
    reminderData.onTimeCount || 0,
    reminderData.lastCompletedAt || ""
  ];

  for (var i = 1; i < data.length; i++) {
    const rowId = String(data[i][0] || "");
    const rowUser = String(data[i][1] || "");
    if (rowId === remId && rowUser === username) {
      remSheet.getRange(i + 1, 1, 1, rowValues.length).setValues([rowValues]);
      return { status: 'success', mode: 'updated' };
    }
  }

  remSheet.appendRow(rowValues);
  return { status: 'success', mode: 'inserted' };
}

function completeReminder(username, reminderData) {
  return saveReminder(username, reminderData);
}
